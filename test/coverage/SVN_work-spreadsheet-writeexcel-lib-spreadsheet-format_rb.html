 390    #
 391    # get_color(colour)
 392    #
 393    # Used in conjunction with the set_xxx_color methods to convert a color
 394    # string into a number. Color range is 0..63 but we will restrict it
 395    # to 8..63 to comply with Gnumeric. Colors 0..7 are repeated in 8..15.
 396    #
 397    def get_color(colour = nil)
 398       # Return the default color, 0x7FFF, if undef,
 399       return 0x7FFF if colour.nil?
 400 
 401       if colour.kind_of?(Numeric)
 402          if colour < 0
 403             return 0x7FFF
 404 
 405          # or an index < 8 mapped into the correct range,
 406          elsif colour < 8
 407             return (colour + 8).to_i
 408 
 409          # or the default color if arg is outside range,
 410          elsif 63 < colour
 411             return 0x7FFF
 412 
 413          # or an integer in the valid range
 414          else
 415             return colour.to_i
 416          end
 417       elsif colour.kind_of?(String)
 418          # or the color string converted to an integer,
 419          if COLORS.has_key?(colour)
 420             return COLORS[colour]
 421 
 422          # or the default color if string is unrecognised,
 423          else
 424             return 0x7FFF
 425          end
 426       else
 427          return 0x7FFF
 428       end
 429    end
